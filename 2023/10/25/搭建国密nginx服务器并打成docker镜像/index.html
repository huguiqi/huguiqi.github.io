FROM centos:centos7.9.2009
ENV LANG=C.UTF-8
RUN yum install -y wget
RUN wget -O /etc/yum.repos.d/CentOS-Base.repo http://mirrors.aliyun.com/repo/Centos-7.repo
RUN yum makecache

# Install tools
RUN yum update -y && \
    yum reinstall -y glibc-common && \
    yum install -y telnet net-tools && \
    yum clean all && \
    rm -rf /tmp/* rm -rf /var/cache/yum/*

docker build -t centos7:test .

启动支持gmssl的nginx服务器

生成服务器自签名相关证书

整理成脚本如下``genCert.sh`：

#!/bin/bash

echo openssl path is: $gmssl

cp $gmssl/ssl/openssl.cnf /usr/local/certs
output=/usr/local/certs/openssl.cnf
echo "# 这个配置必加，否则 gmssl 会报错的。openssl 则无需。
[ v3enc_req ]
basicConstraints = CA:FALSE
keyUsage = keyAgreement, keyEncipherment, dataEncipherment
" >> $output

# 自定义CA证书签发
gmssl ecparam -genkey -name sm2p256v1 -out rootca.key
gmssl req -new -x509 -sm3 -key  rootca.key  -out  rootcasm2.cer -days 10000 -subj '/C=CN/O=HUB/OU=HUGUIQI_SM2'


# Server签名证书
gmssl ecparam -genkey -name sm2p256v1 -noout -out server.key
gmssl req -new -SM3 -key server.key -out server.csr -subj "/C=CN/ST=ShangHai/L=ShangHai/O=zaxh/CN=tom/emailAddress=guiqi_hu@163.com"
gmssl x509 -req -SM3 -days 3650 -in server.csr -extfile openssl.cnf -extensions v3_req -CA rootcasm2.cer -CAkey rootca.key -set_serial 1000000001 -out server.cer


# Server加密证书
gmssl ecparam -genkey -name sm2p256v1 -noout -out server_en.key
gmssl req -new -SM3 -key server_en.key -out server_en.csr -subj "/C=CN/ST=ShangHai/L=ShangHai/O=zaxh/CN=tom/emailAddress=guiqi_hu@163.com"
gmssl x509 -req -SM3 -days 3650 -in server_en.csr -extfile openssl.cnf -extensions v3enc_req -CA rootcasm2.cer -CAkey rootca.key -set_serial 1000002001 -out server_en.cer

# 注意，client的证书主要用来客户端调用和验证gm服务器是否搭建正确
# Client签名证书
gmssl ecparam -genkey -name sm2p256v1 -noout -out client.key
gmssl req -new -key client.key -out client.req -subj "/C=CN/O=HUB/OU=HUGUIQI_SM2/CN=client"
gmssl x509 -req -SM3 -days 3650 -in client.req -extfile openssl.cnf -extensions v3_req -CA rootcasm2.cer -CAkey rootca.key -CAcreateserial  -out client.cer

# client加密证书
gmssl ecparam -genkey -name sm2p256v1 -noout -out client_en.key
gmssl req -new -key client_en.key -out client_en.req -subj "/C=CN/O=HUB/OU=HUGUIQI_SM2/CN=client"
gmssl x509 -req -SM3 -days 3650 -in client_en.req -CA rootcasm2.cer -extfile openssl.cnf -extensions v3enc_req -CAkey rootca.key -CAcreateserial  -out client_en.cer

启动nginx服务器

Dockerfile:

FROM gm-nginx-test:1.0
LABEL maintainer='NGINX Docker Maintainers <guiqi_hu@@163.com>'
LABEL author=huguiqi

COPY index.html /usr/local/nginx/html
COPY nginx.conf /usr/local/nginx/conf/nginx.conf
COPY gmssl-ng2.conf /usr/local/nginx/conf.d/default.conf
RUN mkdir -p /usr/local/certs

WORKDIR /usr/local/certs

RUN rm -rf *

COPY genCert.sh ./
#COPY certs /usr/local/certs
RUN echo 生成证书 && sh ./genCert.sh
RUN ls /usr/local/certs

CMD nginx -g "daemon off;"

gmssl-ng2.conf:

server {

    ssl_certificate      "/usr/local/certs/server.cer";
    ssl_certificate_key  "/usr/local/certs/server.key";
    ssl_certificate      "/usr/local/certs/server_en.cer";
    ssl_certificate_key  "/usr/local/certs/server_en.key";

    listen      443 ssl;
    listen      80;
    server_name localhost;
    keepalive_timeout  70;

    ssl_session_cache    shared:SSL:5m;
    ssl_session_timeout  5m;
    ssl_protocols TLSv1 TLSv1.1 TLSv1.2;
    ssl_ciphers ECDHE-RSA-AES128-GCM-SHA256:ECC-SM4-SM3:ECDHE-SM4-SM3:SM2-WITH-SMS4-SM3:ECDHE-SM2-WITH-SMS4-GCM-SM3:HIGH:!aNULL:!MD5:!RC4:!DHE;
    # ssl_ciphers  HIGH:!aNULL:!MD5;
    ssl_prefer_server_ciphers  on;
    #charset koi8-r;
    #access_log  /var/log/nginx/host.access.log  main;


    location / {
            root   html;
            index  index.html index.htm;
        }
}

version: '2'

docker-compose -f docker-compose-gm.yml up -d --build

下载地址

国密实验室地址：https://www.gmssl.cn/gmssl/index.jsp

国密版curl
基于curl-7.82.0移植，支持单向认证/双向认证。
相关参数:
–gmssl, 启用国密SSL协议
–cert, 指定双向认证签名证书
–key, 指定双向认证签名私钥
–cert2, 指定双向认证加密证书
–key2, 指定双向认证加密私钥

使用示例:
1）单向国密SSL
gmcurl –gmssl -k https://ebssec.boc.cn/
2）证书校验
gmcurl –gmssl –cacert boc.ca.pem https://ebssec.boc.cn/
3）双向国密SSL
gmcurl –gmssl –cert sm2.user01.sig.crt.pem –key sm2.user01.sig.key.pem
–cert2 sm2.user01.enc.crt.pem –key2 sm2.user01.enc.key.pem
–cacert sm2.ca.pem https://demo.gmssl.cn:1443/
注释1：user01相关密钥通过”国密数字证书”生成
注释2：windows下使用’^’符号来分行

运行环境
WindowsXP及以后版本
Centos7/8版本
MacOS/x86_64版本
本地下载: 参见“国密工具下载”。

是客户端和服务端使用gmssl版本不一致导致的，gmssl 基于 openssl 1.1.0d ，client基于新的1.1.1 之后版本，layer层有更改。
回退 client版本可用

我下载的最新的gmcurl版本可能就是基于gmssl 1.1.1之后的版本打包的工具。

解决方式：

1. 只能由官方去彻底解决这个问题了
1. 通过java客户端可以正常调用，不影响服务器之间访问

官方github问题,到目前还是open状态：https://github.com/guanzhi/GmSSL/issues/939
